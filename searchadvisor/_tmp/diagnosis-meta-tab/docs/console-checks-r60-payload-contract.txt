// STEP 1: run this in the live page after the bookmarklet loads.
(async () => {
  const manifestUrl =
    "https://raw.githubusercontent.com/routinefactory/html/main/searchadvisor/_tmp/diagnosis-meta-tab/latest.json?ts=" +
    Date.now();
  const manifest = await fetch(manifestUrl, { cache: "no-store" }).then((r) => r.json());
  const runtimeUrl =
    "https://raw.githubusercontent.com/routinefactory/html/main/searchadvisor/_tmp/diagnosis-meta-tab/" +
    manifest.runtime +
    "?v=" +
    encodeURIComponent(manifest.version) +
    "&ts=" +
    Date.now();
  const runtimeText = await fetch(runtimeUrl, { cache: "no-store" }).then((r) => r.text());
  console.log({
    manifestVersion: manifest.version,
    manifestRuntime: manifest.runtime,
    loadedRuntimeVersion: window.__SEARCHADVISOR_RUNTIME_VERSION__,
    hasApi: !!window.__sadvApi,
    apiReady: !!window.__sadvApi?.isReady?.(),
    hasPayloadContractMarker: runtimeText.includes('window.__SEARCHADVISOR_PAYLOAD_CONTRACT__ = {'),
    hasDecorationHookMarker: runtimeText.includes(
      'window.__SEARCHADVISOR_APPLY_REPORT_DECORATIONS__ = applySnapshotReportDecorations;',
    ),
    hasContractVersionMarker: runtimeText.includes('version: "20260314-payload-contract-v1"'),
  });
})();

// STEP 2: run this in a saved single-account HTML after it opens.
(() => {
  const contract = window.__SEARCHADVISOR_PAYLOAD_CONTRACT__;
  const payload = window.__SEARCHADVISOR_EXPORT_PAYLOAD__;
  const mergedMeta = contract?.getMergedMetaState?.();
  const siteMeta = contract?.getSiteMetaMap?.() || {};
  console.log({
    hasExportPayload: !!payload,
    hasPayloadContract: !!contract,
    contractVersion: contract?.version || null,
    contractMode: contract?.mode || null,
    hasDecorationHook: typeof window.__SEARCHADVISOR_APPLY_REPORT_DECORATIONS__ === "function",
    siteMetaCount: Object.keys(siteMeta).length,
    mergedMetaIsNull: mergedMeta == null,
    currentMode: payload?.curMode || null,
    currentSite: payload?.curSite || null,
    currentTab: payload?.curTab || null,
    allSitesCount: Array.isArray(payload?.allSites) ? payload.allSites.length : null,
  });
})();
